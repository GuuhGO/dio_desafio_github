algoritmo "ContaPar_Vetores"
var
   n: vetor[1..7] de inteiro
   i, quant: inteiro
inicio
   quant <- 0
   Para i <- 1 ate 7 faca
      Escreva("Digite o", i, "° valor: ")
      Leia(n[i])
   FimPara
   Escreval()
   Escreval("Quantidade de numeros pares: ", quant)
   Para i <- 1 ate 7 faca
      Se (n[i] % 2 = 0) entao
         quant <- quant + 1
         Escreval("Posições dos numeros e seus valores: ")
         Escreval("Posicao ", i, " =", n[i])
      FimSe
   FimPara
fimalgoritmo
